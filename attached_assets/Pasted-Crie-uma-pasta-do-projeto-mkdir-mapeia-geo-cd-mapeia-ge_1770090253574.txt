3. Criar a base e as tabelas no PostGIS

Crie a pasta sql/ e o arquivo sql/init.sql:

CREATE EXTENSION IF NOT EXISTS postgis;

CREATE TABLE IF NOT EXISTS properties (
  property_id UUID PRIMARY KEY,
  owner_id UUID NOT NULL,
  name TEXT NOT NULL,
  geom GEOMETRY(MULTIPOLYGON, 4326) NOT NULL,
  area_ha NUMERIC,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE IF NOT EXISTS analyses (
  analysis_id UUID PRIMARY KEY,
  property_id UUID NOT NULL REFERENCES properties(property_id),
  status TEXT NOT NULL,
  datasets_snapshot JSONB,
  results JSONB,
  qa_flags JSONB,
  started_at TIMESTAMPTZ,
  finished_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE IF NOT EXISTS artifacts (
  artifact_id UUID PRIMARY KEY,
  analysis_id UUID NOT NULL REFERENCES analyses(analysis_id),
  type TEXT NOT NULL,
  uri TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE INDEX IF NOT EXISTS idx_properties_geom ON properties USING GIST (geom);


Execute:

psql "postgresql://mapeia:mapeia@localhost:5432/mapeia" -f sql/init.sql


O que você deve ter ao final.

6. Serviço geoespacial mínimo com validação e área correta
6.1 app/services/geo.py

Aqui está o ponto mais crítico. Área precisa ser calculada em projeção métrica.

from shapely.geometry import shape
from shapely.validation import explain_validity
from shapely.ops import unary_union
from pyproj import Transformer

def parse_and_fix(geom_geojson: dict):
    g = shape(geom_geojson)

    if g.is_empty:
        raise ValueError("geometria vazia")

    if not g.is_valid:
        msg = explain_validity(g)
        g2 = g.buffer(0)
        if not g2.is_valid:
            raise ValueError(f"geometria invalida: {msg}")
        g = g2

    if g.geom_type == "Polygon":
        g = unary_union([g])

    if g.geom_type != "MultiPolygon":
        raise ValueError(f"tipo geometrico inesperado: {g.geom_type}")

    return g

def area_hectares_wgs84(geom_multipolygon):
    """
    Calcula area em hectares reprojectando para uma projeção métrica local.
    Regra prática. usar UTM pelo centróide do polígono.
    """
    lon, lat = geom_multipolygon.centroid.x, geom_multipolygon.centroid.y
    zone = int((lon + 180) / 6) + 1
    epsg = 32700 + zone if lat < 0 else 32600 + zone

    transformer = Transformer.from_crs("EPSG:4326", f"EPSG:{epsg}", always_xy=True)

    def _tx(x, y, z=None):
        return transformer.transform(x, y)

    geom_proj = type(geom_multipolygon)([poly for poly in geom_multipolygon.geoms])
    geom_proj = shapely_transform(geom_multipolygon, _tx)

    area_m2 = geom_proj.area
    return area_m2 / 10000.0

def shapely_transform(geom, func):
    try:
        from shapely.ops import transform
        return transform(func, geom)
    except Exception as e:
        raise RuntimeError(f"falha ao transformar geometria: {e}")


O que você deve ter ao final.
. Uma função que recebe GeoJSON, corrige, e calcula área em hectares de forma consistente.

from shapely.geometry import mapping

def compute_basic_metrics(geom, area_ha: float):
    minx, miny, maxx, maxy = geom.bounds
    cx, cy = geom.centroid.x, geom.centroid.y

    results = {
        "property": {
            "area_ha": float(area_ha),
            "centroid": {"lon": float(cx), "lat": float(cy)},
            "bbox": {"minx": float(minx), "miny": float(miny), "maxx": float(maxx), "maxy": float(maxy)},
        },
        "intersections": {
            "ucs": [],
            "tis": [],
        },
        "compliance": {
            "flags": [],
        },
        "provenance": {
            "datasets": [],
        }
    }
    return results

8. Worker Celery que processa análises
8.1 app/tasks.py
import os
import json
from celery import Celery
from datetime import datetime, timezone
from sqlalchemy import text
from app.db import SessionLocal

celery_app = Celery(
    "mapeia_workers",
    broker="redis://localhost:6379/0",
    backend="redis://localhost:6379/0",
)

@celery_app.task(name="run_analysis")
def run_analysis(analysis_id: str):
    from app.services.geo import parse_and_fix, area_hectares_wgs84
    from app.services.metrics import compute_basic_metrics
    from app.services.report import build_pdf_report

    db = SessionLocal()
    try:
        db.execute(text("UPDATE analyses SET status='running', started_at=now() WHERE analysis_id=:id"), {"id": analysis_id})

        row = db.execute(
            text("""
                SELECT a.analysis_id, a.property_id, p.name, p.geom
                FROM analyses a
                JOIN properties p ON p.property_id = a.property_id
                WHERE a.analysis_id = :id
            """),
            {"id": analysis_id},
        ).mappings().first()

        geom_geojson = json.loads(row["geom"])
        geom = parse_and_fix(geom_geojson)
        area_ha = area_hectares_wgs84(geom)

        results = compute_basic_metrics(geom, area_ha)

        pdf_path = build_pdf_report(row["property_id"], analysis_id, row["name"], results)

        db.execute(
            text("UPDATE analyses SET status='done', finished_at=now(), results=:res WHERE analysis_id=:id"),
            {"id": analysis_id, "res": json.dumps(results)},
        )
        db.execute(
            text("INSERT INTO artifacts (artifact_id, analysis_id, type, uri) VALUES (gen_random_uuid(), :aid, 'report_pdf', :uri)"),
            {"aid": analysis_id, "uri": pdf_path},
        )
        db.commit()

    except Exception as e:
        db.execute(text("UPDATE analyses SET status='failed', finished_at=now(), qa_flags=:q WHERE analysis_id=:id"),
                   {"id": analysis_id, "q": json.dumps({"error": str(e)})})
        db.commit()
        raise
    finally:
        db.close()


Nota.
. gen_random_uuid() exige extensão pgcrypto.
. Se preferir, eu ajusto para UUID no Python. Mais simples no começo.

from pathlib import Path
from reportlab.lib.pagesizes import A4
from reportlab.pdfgen import canvas

def build_pdf_report(property_id: str, analysis_id: str, property_name: str, results: dict):
    out_dir = Path("tmp")
    out_dir.mkdir(exist_ok=True)
    pdf_path = out_dir / f"mapeia_{property_id}_{analysis_id}.pdf"

    c = canvas.Canvas(str(pdf_path), pagesize=A4)
    w, h = A4

    c.setFont("Helvetica-Bold", 14)
    c.drawString(40, h - 50, "MapeIA. Relatorio de Vistoria. Analise Automatica")

    c.setFont("Helvetica", 10)
    c.drawString(40, h - 80, f"Imovel: {property_name}")
    c.drawString(40, h - 95, f"Area (ha): {results['property']['area_ha']:.2f}")

    cx = results["property"]["centroid"]["lon"]
    cy = results["property"]["centroid"]["lat"]
    c.drawString(40, h - 110, f"Centroide: {cx:.6f}, {cy:.6f}")

    c.setFont("Helvetica-Bold", 11)
    c.drawString(40, h - 140, "Resumo")
    c.setFont("Helvetica", 10)
    c.drawString(40, h - 160, "Camadas legais e uso do solo serao adicionados na proxima etapa do pipeline.")

    c.showPage()
    c.save()

    return str(pdf_path)

10.2 app/main.py
import json
import uuid
from fastapi import FastAPI, Depends, HTTPException
from sqlalchemy import text
from sqlalchemy.orm import Session
from app.db import SessionLocal
from app.schemas import PropertyCreate, AnalysisCreate
from app.tasks import celery_app

app = FastAPI(title="MapeIA Geo API", version="1.0.0")

def get_db():
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

@app.post("/v1/properties")
def create_property(payload: PropertyCreate, db: Session = Depends(get_db)):
    pid = uuid.uuid4()
    db.execute(
        text("INSERT INTO properties (property_id, owner_id, name, geom) VALUES (:pid, :oid, :name, :geom)"),
        {"pid": pid, "oid": payload.owner_id, "name": payload.name, "geom": json.dumps(payload.geom_geojson)},
    )
    db.commit()
    return {"property_id": str(pid)}

@app.post("/v1/properties/{property_id}/analyses")
def create_analysis(property_id: str, payload: AnalysisCreate, db: Session = Depends(get_db)):
    aid = uuid.uuid4()
    db.execute(
        text("INSERT INTO analyses (analysis_id, property_id, status, datasets_snapshot) VALUES (:aid, :pid, 'queued', :ds)"),
        {"aid": aid, "pid": property_id, "ds": json.dumps(payload.datasets or {"defaults": True})},
    )
    db.commit()

    celery_app.send_task("run_analysis", args=[str(aid)])
    return {"analysis_id": str(aid), "status": "queued"}

@app.get("/v1/analyses/{analysis_id}")
def get_analysis(analysis_id: str, db: Session = Depends(get_db)):
    row = db.execute(
        text("SELECT analysis_id, status, results, qa_flags FROM analyses WHERE analysis_id=:id"),
        {"id": analysis_id},
    ).mappings().first()

    if not row:
        raise HTTPException(status_code=404, detail="analysis not found")

    return {
        "analysis_id": str(row["analysis_id"]),
        "status": row["status"],
        "results": row["results"],
        "qa_flags": row["qa_flags"],
    }
